function openDetail(id){
  const p=products[id];if(!p)return;

  document.getElementById('detailName').textContent=p.name;
  document.getElementById('detailPrice').textContent=p.price;
  document.getElementById('detailOldPrice').textContent=p.oldPrice||'';
  document.getElementById('detailDesc').textContent=p.desc;
  const iconEl=document.getElementById('detailIcon');
  iconEl.setAttribute('data-lucide',p.icon||'camera');
  const bwrap=document.getElementById('detailBadgeWrap');
  bwrap.innerHTML=p.badge?`<span class="card-badge ${p.badge==='hot'?'hot':p.badge==='sale'?'sale':''}">${p.badgeLabel}</span>`:'';
  const specsEl=document.getElementById('detailSpecs');
  if(p.specs&&p.specs.length){
    specsEl.innerHTML=p.specs.map(([k,v])=>`<div class="spec-row"><span style="color:#94a3b8">${k}</span><span style="font-weight:600;color:#0f172a">${v}</span></div>`).join('');
    specsEl.style.display='block';
  }else{specsEl.style.display='none';}
  document.getElementById('detailAddBtn').onclick=()=>{addToCart(p.name,p.price);closeDetail();};
  document.getElementById('detailOverlay').classList.add('open');
  lucide.createIcons();
}
